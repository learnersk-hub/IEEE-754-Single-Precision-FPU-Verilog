`timescale 1ns/1ps

module fp_div_tb;

reg [31:0] a;
reg [31:0] b;

wire [31:0] result;

fp_div uut(
    .a(a),
    .b(b),
    .result(result)
);

initial begin

    $dumpfile("fp_div.vcd");
    $dumpvars(0,fp_div_tb);

    // 8 / 2 = 4

    a = 32'h41000000;
    b = 32'h40000000;

    #20;

    $display("8/2 = %h",result);

    // -8 / 2 = -4

    a = 32'hC1000000;
    b = 32'h40000000;

    #20;

    $display("-8/2 = %h",result);

    $finish;

end

endmodule
